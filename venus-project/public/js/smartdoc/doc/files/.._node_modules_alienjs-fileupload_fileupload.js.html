    var $ = require('jquery');
var Widget = require('lib/widget/widget.js');
require('widget/fileupload/fileupload.css');


var defaults = {
    $el   : '',
    url   : '',
    preview : '.fileupload-preview',
    uploadBtn : '.fileupload-btn',
    submitBtn : '.fileupload-submit',
    source : [],
    max :  5,
    message : function(msg){
        window.alert(msg)
    },
    onComplete : function( data ){},
    onError : function( data ){},
    onSuccess : function( data ){},
    onDeleteComplete : function( data ){},
    onDeleteError : function( data ){},
    onDeleteSuccess : function( data ){}
}

/**
 * html5 ajax多文件上传
 * @module 273
 * @FileUpload 273
 * @params $el {string|object} 元素，选择器
 * @params url {string} 提交地址
 * @params preview {string} 预览显示容器，选择器
 * @params submitBtn {string} 上传按钮，选择器 。
 * @params uploadBtn {string} 选择文件按钮，选择器 。 当submitBtn 为空时 uploadBtn
 * @params source {array} 初始数据 [{ name : '1.jpg' , src : '1.jpg'}]
 * @params onComplete {function} 上传完成，无论成功失败
 * @params onError {function} 上传失败，无论成功失败
 * @params onSuccess {function} 上传成功，无论成功失败
 */
var FileUpload = function( options ){
    this.options = $.extend({} , defaults  , options );
    this.$elem = this.options.$el = this.options.$el.length ? this.options.$el : $( this.options.el );
    this.$preview = $(this.options.preview , this.$elem );
    this.$submit = $(this.options.submitBtn , this.$elem );
    this.files = {} ;
    this.init();
};

FileUpload.prototype = {
    constructor : FileUpload,
    init : function(){
        var self = this;
        this.deferred = $.Deferred();
        this.$file = this.$elem.find(':file').off('change.fileupload').on('change.fileupload',function(e){
            self.add(  e.target.files );
        })
        this.$preview.off('click.fileupload').on('click.fileupload','.icon-del',function(e){
            self.remove( this );
        }).on('click.fileupload','.icon-error',function(e){
            self.upload( self.files[ $(this).closest('li').data('id')] );
        });
        this.$submit.off('click.fileupload').on('click.fileupload',function(e){
            self.uploadAll();
        });
        this.add(this.options.source);
        return this;
    },
    add: function( files ){
        var self = this;
        for(var i = 0; i < files.length; i++ ){
            if( files.length >= this.options.max ){
                this.options.message('上传文件不得超过' + this.options.max + '张');
                return this ;
            }

            var file = files[i], id = parseInt(Math.random() * 1000) + new Date().getTime();;
            file.id = id ;
            file.deferred = $.Deferred();
            self.files[ id ] = file ;

            if( file.size ) {
                if (window.URL) {
                    //File API
                    //alert(files[0].name + "," + files[0].size + " bytes");
                    var $img = $('<img>').attr('src', window.URL.createObjectURL(file)).on('load', function () {
                        window.URL.revokeObjectURL(this.src); //图片加载后，释放object URL
                    });
                    self.render(file, $img);
                } else if (window.FileReader) {
                    //opera不支持createObjectURL/revokeObjectURL方法。我们用FileReader对象来处理
                    var reader = new FileReader();
                    reader.readAsDataURL(files[0]);
                    reader.onload = function (e) {
                        //alert(files[0].name + "," +e.total + " bytes");
                        var $img = $('<img>').attr('src', this.result);
                        self.render( file, $img);
                    }
                }
                if (!this.$submit.length) {
                    this.upload( file );
                }
            }
            else{
                self.render( file, $('<img>').attr('src' , file.src ));
            }
        }
        return this;
    },
    remove :function( elem , url){
        var self = this;
        var $item = $(elem).closest('li'),  id = $item.data('id'),
            file = self.files[ id ];
        url = url || this.options.url;
        var fd = new FormData(), name = $item.find(':hidden').val();
        if( file )
            this.post( fd , url + '?id=' + name , file , function( data ){
                //self.files.splice( id, 1 );
                delete self.files[id] ;
                $item.remove();
            }, 'delete');
        else
            $item.remove();
        return this;
    },
    render : function( file , img ){
        var $li = $('<li/>').attr('data-id' , file.id ),
            $item = $('<div class="fileupload-item"></div>'),
            $loading = $('<div class="uploading"></div>'),
            $progress = $('<div class="progress"><i></i></div>'),
            $complete = $('<i class="icon-success"></i>'),
            $error = $('<a class="icon-error" href="javascript:;">上传失败,点击重新上传</a>'),
            $errorDelete = $('<i class="icon-error-del">删除失败</i>'),
            $input = $('<input type="hidden" name="file">'),
            $del = $('<a href="javascript:;" class="icon-del"></a>');
        $li.append( $item.append( img ))
            .append($del)
            .append($loading)
            .append($complete)
            .append($error)
            .append($errorDelete)
            .append($input)
            .append($progress);

        if( img.width () > img.height() ){
            img.css({
                "margin-top" : ( $li.height() - img.height()) / 2
            })
        }
        else{
            img.css({
                height: "100%",
                width : 'auto'
            })
        }
        $li.appendTo(this.$preview);
    },
    uploadAll: function( url ){
        for( var i in this.files){
            this.files[i].deferred && this.upload( this.files[i] ,url  );
        }
        return this;
    },
    post : function( data , url , file, callback ,type ){
        var self = this;
        var xhr = new XMLHttpRequest(),$item = self.$preview.find('li[data-id='+ file.id +']');

        function error ( data ){

            file.deferred.rejectWith(data);
            if( type == 'upload'){
                $(this).addClass('error');
                self.options.onError && self.options.onError.call( this ,data);
            }
            else if ( type == 'delete'){
                $(this).addClass('error-del')
                self.options.onDeleteError && self.options.onDeleteError.call( $item , data);
            }
            delete self.files[ file.id] ;
        }

        xhr.upload.addEventListener("progress", function(e){
            if (e.lengthComputable) {
                var percentComplete = Math.round(e.loaded * 100 / e.total);
                var $progress =  $item.find('.progress i').css('width', percentComplete.toString() + '%');
                if( percentComplete == 100 )
                    $progress.hide()
            }
            else {
                //document.getElementById('progressNumber').innerHTML = 'unable to compute';
            }
        }, false);
        xhr.addEventListener("load", function(){
            var data = new Function('return '+  xhr.responseText )();

            if( data.error == 0 ){
                $item.remove('error').addClass('success').find('img').attr('src' , data.url );
                $item.find(':hidden').val( data.name );
                file.deferred.resolveWith(data);

                if( type == 'upload')
                    self.options.onSuccess && self.options.onSuccess.call( $item , data);
                else if ( type == 'delete')
                    self.options.onDeleteSuccess && self.options.onDeleteSuccess.call( $item , data);

                self.isCompleted() && self.deferred.resolveWith(data);
                callback && callback.call( this , data );
            }
            else{
                error.call( $item , data );
            }
            if( type == 'upload')
                self.options.onComplete && self.options.onComplete.call( $item , data);
            else if ( type == 'delete')
                self.options.onDeleteComplete && self.options.onDeleteComplete.call( $item , data);
        }, false);
        xhr.addEventListener("error", function(e){
            var data = new Function('return '+  xhr.responseText )();
            error.call( $item , data );
            self.options.onComplete && self.options.onComplete.call( $item , data);
        }, false);
        xhr.addEventListener("abort", function(e){
            var data = new Function('return '+  xhr.responseText )();
            error.call( $item , data );
            self.options.onComplete && self.options.onComplete.call( $item , data);
        }, false);
        xhr.open("POST",  url );
        xhr.send( data );
    },
    upload : function( file , url ){
        var self = this;
        url = url || this.options.url;
        var fd = new FormData();
        fd.append("file", file );
        this.post( fd , url  , file, function( data ){

        }, 'upload');
        return file.deferred.promise();
    },
    isCompleted : function(){
        for(var i = 0 ; i < this.files.length; i++ ){
            if( this.files.deferred && this.files.deferred.state() != 'resolved')
                return false;
        }
        return true;
    }
}

FileUpload.init = function( options ){
    return new FileUpload(options);
}

module.exports = FileUpload;
