    CREATE DEFINER = `root`@`localhost` FUNCTION `GetGuid`()
    RETURNS varchar(40)
    BEGIN
    DECLARE id VARCHAR(40);
    SET id=(SELECT Replace(uuid(),'-',''));
    return id;
    END;
