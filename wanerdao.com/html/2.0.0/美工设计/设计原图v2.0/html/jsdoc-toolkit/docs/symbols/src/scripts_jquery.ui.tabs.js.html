  3 ;(function($) {
  4     /**
  5      * @author caizhongyi
  6      * @version 1.0
  7      * @description 切换卡效果
  8      * @requires  jquery.1.7.2
  9      * @requires  jquery.core
 10      * @class  tabs
 11      * @params {string} selector
 12      * @params {object} options
 13      * @return {object} dialog
 14      * @example new $.ui.tabs('.tabs');
 15      * */
 16   $.ui.tabs = function(selector,options) {
 17        var _options = {
 18             effect			: 	"fade",		// x || y || fade || img || normal
 19             event			: 	"click", 	// mouseover or click or hover
 20             auto			: 	false, 		// 自动
 21             index			: 	0,         	// 当前index
 22 			tempIndex		:   0,
 23 			easing			:	"linear",  	//linear" 和 "swing  || easeOutExpo
 24 			duration		:   'normal',
 25 			offset			:   1,
 26 			cssSelected		:   'selected', //当前选种样式
 27 			reversion		:	true,		 //是否回滚
 28             interval		: 	3000,
 29 			loadCallback	:   null,       //用于动态载入
 30 			changedCallback	:   null,
 31 			//scrollbar		:   $("#scrollbar").mCustomScrollbar("vertical",400,"easeOutCirc",1.05,"auto","yes","yes",10),
 32 			widget	 		:	{
 33 				view		:	 '.tabs-view',   //大图片显示区域
 34 				nav			:	 '.tabs-nav',
 35 				panel 		: 	 '.tabs-panel',	 //图片滚动区域
 36 				prev    	:    '.tabs-prev',
 37 				next		:    '.tabs-next',
 38 				clip		:	 '.tabs-clip'
 39 			}
 40         }
 41 		this.extend(selector , _options , options);
 42         return this;
 43     }
 44 	$.ui.tabs.extend({
 45 		init	: 	function(){
 46 			var _this = this;
 47 			// ui
 48 			this.$clip.css({ overflow : 'hidden'});
 49 			// load 
 50 			this.update();
 51 			// event 
 52 			this.$next.unbind('click.tabs').bind('click.tabs',function(){ _this.next();});
 53 			this.$prev.unbind('click.tabs').bind('click.tabs',function(){ _this.prev();});
 54 			this.$nav.undelegate('click.tabs').delegate(this.$nav.children('li').length ? 'li' : 'a' ,this.options.event + '.tabs',function(){
 55 				_this.goto($(this).index());
 56 			});
 57 			this.$panel.undelegate('click.tabs').delegate(this.$panel.children('li').length ? 'li' : 'a'  ,this.options.event + '.tabs',function(){_this.view($(this))})
 58 						.unbind('mouseenter.tabs mouseleave.tabs').bind({
 59 							'mouseenter.tabs' : function(){_this.stop()},
 60 							'mouseleave.tabs' : function(){_this.options.auto && _this.start()}
 61 						})
 62 			
 63 			//初始化
 64 			this.show(this.index());
 65 			if(this.options.auto){
 66 				
 67 				this.start();
 68 			}
 69 			return this;
 70 		  
 71 		},
 72 		update	  : function(){
 73 			var attr = {};
 74 			var items = this.$panel.children(),
 75                 count = items.length;
 76 			switch(this.options.effect){
 77 				case 'x' : attr = { width : (items.css({ 'float' : 'left'}).outerWidth() + parseFloat(items.css('margin-left'))   + parseFloat(items.css('margin-right')) ) * count } ;  break;
 78 				case 'y' : attr = { height : (items.outerHeight() + parseFloat(items.css('margin-top'))  + parseFloat(items.css('margin-bottom')) ) * count} ; break;
 79 				default : break;
 80 			}
 81 			attr.position = 'relative';
 82 			this.$panel.css(attr);
 83 			return this;
 84 		},
 85 		//获取大图片的地址
 86 		bigImage : function(){
 87 			return $('img',this.$nav).eq(this.index());
 88 		},
 89 		length  : function(){
 90 			return this.pageCount();
 91 		},
 92         pageCount : function(){
 93             var length = this.options.effect == 'fade' ? this.$nav.children().length : this.$panel.children().length;
 94             var val = parseInt( length / this.options.offset);
 95             return length % this.options.offset > 0 ? val + 1 : val;
 96         },
 97 		offset : function(){
 98 			var offset = 0;
 99 			var items = this.$panel.children();
100 			switch(this.options.effect){
101 				case 'x' : offset = items.outerWidth() + parseFloat(items.css('margin-left'))  + parseFloat(items.css('margin-right')) ; offset = offset * this.options.offset; break;
102 				case 'y' : offset = items.outerHeight() + parseFloat(items.css('margin-top'))  + parseFloat(items.css('margin-bottom')) ; offset = offset * this.options.offset; break;
103 				default  : break;
104 			}
105 			return offset;
106 		},
107 		load  : function(index , callback){
108 			var _this = this;
109 			if((this.options.effect != 'x' && this.options.effect != 'y') || ((this.options.effect == 'x' || this.options.effect == 'y') && index >= this.length())){
110 				if(this.options.loadCallback){
111 					this.options.loadCallback({
112 						target			: this,
113 						index 			: this.index() ,
114 						$currentTarget	: this.$nav.children(':eq('+ index +')'),
115 						count			: this.length()
116 					},this.$panel , callback);
117 				}
118 				else{
119 					callback && callback();
120 				}
121 			}
122 			else{
123 				callback && callback();
124 			}
125 			return this;
126 		},
127 		start : function()
128 		{
129 			var _this = this;
130 			if(!this.timer){
131 				this.timer = setInterval(function(){_this.next()},_this.options.interval);
132 			}
133 			return this;
134 		},
135 		stop : function()
136 		{
137 			clearInterval(this.timer);
138 			this.timer = null;
139 		},
140 		view	: function($elem){
141 			var image = $elem;
142 			$('img',this.$view).attr({'src' : image.attr('data-url') ,'title' : image.attr('title')}).stop(true,true).hide().fadeIn();
143 			return this;
144 		},
145 		show	: function(index){
146 			var _this = this;
147 
148 			this.load(index , function(){
149 				_this.index(index);
150 				var $navItems = _this.$nav.children();
151 				$navItems.eq(_this.index()).addClass(_this.options.cssSelected);
152 				if(_this.index() != _this.tempIndex())
153 					$navItems.eq(_this.tempIndex()).removeClass(_this.options.cssSelected);
154 
155 				switch(_this.options.effect){
156 					case 'normal': _this.$panel.children(':eq('+_this.index()+')').show();break;
157 					case 'fade' : _this.animate(_this.$panel.children(':eq('+_this.index()+')').stop(true,true) , { opacity : 'show'}); break;
158 					case 'x' 	: _this.animate(_this.$panel.stop() , { left : -_this.offset() * _this.index()}); break;
159 					case 'y' 	: _this.animate(_this.$panel.stop() , { top : -_this.offset() * _this.index()}); break;
160 					default  	: 
161 						var image = _this.bigImage();
162 						$('img',_this.$panel).attr({'src' : image.attr('data-url') ,'title' : image.attr('title')}).stop(true,true).hide().fadeIn();
163 						break;
164 				}
165 				_this.options.changedCallback && _this.options.changedCallback({
166 					target  : _this, 
167 					$target : _this.$,
168 					$panel	: _this.$panel,
169 					index	: _this.index(),
170 					offset  : _this.offset()
171 					
172 				})
173 				_this.view(_this.$panel.children().eq(_this.index()));
174 			});
175 			return this;
176 		},
177 		hide	: function(index){
178 			switch(this.options.effect){
179 				case 'normal': this.$panel.children(':eq('+this.tempIndex()+')').hide();break;
180 				case 'fade' : this.animate(this.$panel.children(':eq('+this.tempIndex()+')').stop(true,true) , { opacity : 'hide'}); break;
181 				//case 'x' : this.animate(this.$panel , { left : this.offset() * index}); break;
182 				//case 'y' : this.animate(this.$panel , { top : this.offset() * index}); break;
183 				default  :  break;
184 			}
185 			return this;
186 		}
187 		
188 	}).extend($.ui.switchObject.prototype);
189 
190 }(jQuery));
