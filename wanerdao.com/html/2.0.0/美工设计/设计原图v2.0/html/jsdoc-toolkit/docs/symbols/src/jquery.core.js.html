 20 Function.prototype.extend = function (superClass) {
 21     if (typeof superClass === 'function') {//类式继承
 22         var F = function () {
 23         }; //创建一个中间函数对象以获取父类的原型对象
 24         F.prototype = superClass.prototype; //设置原型对象
 25         this.prototype = new F(); //实例化F, 继承父类的原型中的属性和方法，而无需调用父类的构造函数实例化无关的父类成员
 26         this.prototype.constructor = this; //设置构造函数指向自己
 27         this.superClass = superClass; //同时，添加一个指向父类构造函数的引用，方便调用父类方法或者调用父类构造函数
 28     } else if (typeof superClass === 'object') { //方法的扩充
 29         var pro = this.prototype;
 30         for (var k in superClass) {
 31             if (!pro[k]) { //如果原型对象不存在这个属性，则复制
 32                 pro[k] = superClass[k];
 33             }
 34         }
 35     } else {
 36         throw new Error('fatal error:"Function.prototype.extend" expects a function or object');
 37     }
 38 
 39     return this;
 40 
 41 };
