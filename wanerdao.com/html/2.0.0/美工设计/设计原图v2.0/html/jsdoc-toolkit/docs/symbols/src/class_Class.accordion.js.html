 17 ;(function ($) {
 18     /**
 19      * @author caizhongyi
 20      * @version 1.0
 21      * @description 手风琴
 22      * @requires  jquery.1.7.2
 23      * @class  accordion
 24      * @params {string} selector
 25      * @params {object} options
 26      * @params {string} [options.effect=horizontal]
 27      * @params {string} [options.header=h3]
 28      * @params {string} [options.panel=div]
 29      * @params {string} [options.event=click]
 30      * @params {string} [options.selected=selected]
 31      * @params {string} [options.index=0]
 32      * @params {string} [options.collapsible=true]
 33      * @return {object} accordion对象
 34      * @example
 35     * */
 36     Class.accordion = Class.get(
 37         /**
 38          * @lends Class.accordion
 39          * */
 40         {
 41         index         : 0,
 42         tempIndex    : 0,
 43         EVENT_CHANGE:'change',
 44         initialize : function(selector , options){
 45             var obj = $(this);
 46             var _options = {
 47                 effect : 'horizontal',
 48                 header : 'h3',
 49                 panel : 'div',
 50                 event  : 'click',
 51                 selected :"selected",
 52                 index : 0 ,
 53                 collapsible : true
 54             };
 55             this.effects = Class.accordion.effects;
 56             this.options = $.extend(true,{},_options, options);
 57             var _this = this;
 58             this.$ = $(selector);
 59             var index = this.options.index,tempIndex ;
 60             var event = this.options.event + '.accordion';
 61 
 62             this.update();
 63             this.$panels.hide();
 64 
 65             this.show(this.index);
 66 
 67             this.$.off(event).on(event,this.options.header,function(){
 68                 index = $(this).index(_this.options.header);
 69                 _this.show(index);
 70             });
 71 
 72         },
 73         update : function(){
 74             this.$panels =  this.$.children(this.options.panel);
 75             this.$headers = this.$.children(this.options.header);
 76             return this;
 77         },
 78         show : function(index)
 79         {
 80             if(index != this.tempIndex ){
 81                 $.proxy(this.effects[this.options.effect].show,this.$panels.eq(index))();
 82                // this.effects[this.options.effect].show(this.$panels.eq(index));
 83                 this.$headers.eq(index).addClass(this.options.selected);
 84                 this.hide(this.index);
 85                 this.tempIndex = this.index;
 86                 this.index = index;
 87             }
 88             else{
 89                 if(this.options.collapsible){
 90                     this.toggle(index);
 91                 }
 92             };
 93             return this;
 94         },
 95         hide : function(index)
 96         {
 97             $.proxy(this.effects[this.options.effect].hide,this.$panels.eq(index))();
 98             //this.effects[this.options.effect].hide(this.$panels.eq(index));
 99             this.$headers.eq(index).removeClass(this.options.selected);
100             return this;
101         },
102         toggle : function(index)
103         {
104             if(this.$panels.eq(index).is(':hidden')){
105                 $.proxy(this.effects[this.options.effect].show,this.$panels.eq(index))();
106             }
107             else{
108                 this.hide(index)
109             };
110             return this;
111         }
112     });
113 
114     /**
115      * @memberOf Class.accordion
116      * */
117     Class.accordion.effects = {
118         horizontal : {
119             show : function(){
120                 this.stop().animate({ height : 'show'});
121             },
122             hide : function(){
123                 this.stop().animate({ height : 'hide'});
124             }
125         },
126         vertical : {
127             show : function(){
128                 this.stop().animate({ width : 'show'});
129             },
130             hide : function(){
131                 this.stop().animate({ width : 'hide'});
132             }
133         },
134         'default' : {
135             show : function(){ this.show();},
136             hide : function(){ this.hide();}
137         }
138     };

140 })(jQuery);
